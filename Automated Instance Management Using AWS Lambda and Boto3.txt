import boto3

def lambda_handler(event, context):
    ec2 = boto3.client('ec2')

    # -------- Auto-Stop --------
    stop_response = ec2.describe_instances(
        Filters=[
            {'Name': 'tag:Action', 'Values': ['Auto-Stop-jsree']},
            {'Name': 'instance-state-name', 'Values': ['running']}
        ]
    )

    stop_instance_ids = [
        instance['InstanceId']
        for reservation in stop_response['Reservations']
        for instance in reservation['Instances']
    ]

    if stop_instance_ids:
        ec2.stop_instances(InstanceIds=stop_instance_ids)
        print("Stopped instances:", stop_instance_ids)
    else:
        print("No Auto-Stop instances found")

    # -------- Auto-Start --------
    start_response = ec2.describe_instances(
        Filters=[
            {'Name': 'tag:Action', 'Values': ['Auto-Start-jsree']},
            {'Name': 'instance-state-name', 'Values': ['stopped']}
        ]
    )

    start_instance_ids = [
        instance['InstanceId']
        for reservation in start_response['Reservations']
        for instance in reservation['Instances']
    ]

    if start_instance_ids:
        ec2.start_instances(InstanceIds=start_instance_ids)
        print("Started instances:", start_instance_ids)
    else:
        print("No Auto-Start instances found")

    return {
        "statusCode": 200,
        "body": "EC2 Auto Start/Stop executed"
    }
